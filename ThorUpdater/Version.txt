 ldDate,;
 lcappName,;
 lcmajorVersion,;
 lcTemp

*lnJulian = val(sys(11, m.ldDate)) - val(sys(11, {^2000-01-01}))
*lcJulian = padl(transform(lnJulian), 4, '0')

lcTemp = m.lcappName + ' - ' + m.lcmajorVersion + ;
	' - update - ' + dtoc(m.ldDate, 1)

AddProperty(toUpdateObject, 'Notes', m.lcTemp)
